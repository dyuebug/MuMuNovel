```python
async def update_item(
    db: AsyncSession,
    item_id: str,
    item_data: ItemUpdate,
    user_id: str
):
    try:
        item = await db.get(Item, item_id)
        if not item:
            raise HTTPException(status_code=404, detail="项目不存在")

        if item.user_id != user_id:
            raise HTTPException(status_code=403, detail="无权限")

        # 只更新提供的字段
        update_data = item_data.model_dump(exclude_unset=True)
        for key, value in update_data.items():
            setattr(item, key, value)

        await db.commit()
        await db.refresh(item)
        return item
    except Exception as e:
        await db.rollback()
        raise
```

### 2. 分页查询

```python
from typing import List, Generic, TypeVar
from pydantic import BaseModel

T = TypeVar('T')

class PaginatedResponse(BaseModel, Generic[T]):
    items: List[T]
    total: int
    page: int
    page_size: int
    total_pages: int

async def get_paginated_items(
    db: AsyncSession,
    user_id: str,
    page: int = 1,
    page_size: int = 10
) -> PaginatedResponse[ItemResponse]:
    # 计算偏移量
    offset = (page - 1) * page_size

    # 查询总数
    count_result = await db.execute(
        select(func.count(Item.id)).where(Item.user_id == user_id)
    )
    total = count_result.scalar()

    # 查询数据
    result = await db.execute(
        select(Item)
        .where(Item.user_id == user_id)
        .offset(offset)
        .limit(page_size)
    )
    items = result.scalars().all()

    return PaginatedResponse(
        items=items,
        total=total,
        page=page,
        page_size=page_size,
        total_pages=(total + page_size - 1) // page_size
    )
```

### 3. 事务处理

```python
from sqlalchemy.exc import IntegrityError

async def create_project_with_characters(
    db: AsyncSession,
    project_data: ProjectCreate,
    characters_data: List[CharacterCreate],
    user_id: str
):
    """创建项目并同时创建角色（事务）"""
    try:
        # 创建项目
        project = Project(user_id=user_id, **project_data.model_dump())
        db.add(project)
        await db.flush()  # 获取项目 ID，但不提交

        # 创建角色
        characters = []
        for char_data in characters_data:
            character = Character(
                project_id=project.id,
                **char_data.model_dump()
            )
            db.add(character)
            characters.append(character)

        # 提交事务
        await db.commit()
        await db.refresh(project)

        return project
    except IntegrityError as e:
        await db.rollback()
        raise HTTPException(status_code=400, detail="数据完整性错误")
    except Exception as e:
        await db.rollback()
        raise
```

## AI 服务集成

### 1. AI Provider 基类

```python
from abc import ABC, abstractmethod
from typing import AsyncGenerator

class BaseAIProvider(ABC):
    """AI 提供商基类"""

    def __init__(self, api_key: str, base_url: str = None):
        self.api_key = api_key
        self.base_url = base_url

    @abstractmethod
    async def generate(
        self,
        prompt: str,
        max_tokens: int = 1000,
        temperature: float = 0.7,
        **kwargs
    ) -> str:
        """生成文本"""
        pass

    @abstractmethod
    async def generate_stream(
        self,
        prompt: str,
        max_tokens: int = 1000,
        temperature: float = 0.7,
        **kwargs
    ) -> AsyncGenerator[str, None]:
        """流式生成文本"""
        pass
```

### 2. OpenAI Provider 实现

```python
import openai
from typing import AsyncGenerator

class OpenAIProvider(BaseAIProvider):
    """OpenAI 提供商"""

    def __init__(self, api_key: str, base_url: str = None):
        super().__init__(api_key, base_url)
        self.client = openai.AsyncOpenAI(
            api_key=api_key,
            base_url=base_url
        )

    async def generate(
        self,
        prompt: str,
        max_tokens: int = 1000,
        temperature: float = 0.7,
        model: str = "gpt-4o-mini",
        **kwargs
    ) -> str:
        """生成文本"""
        try:
            response = await self.client.chat.completions.create(
                model=model,
                messages=[{"role": "user", "content": prompt}],
                max_tokens=max_tokens,
                temperature=temperature,
                **kwargs
            )
            return response.choices[0].message.content
        except Exception as e:
            logger.error(f"OpenAI 生成失败: {e}")
            raise

    async def generate_stream(
        self,
        prompt: str,
        max_tokens: int = 1000,
        temperature: float = 0.7,
        model: str = "gpt-4o-mini",
        **kwargs
    ) -> AsyncGenerator[str, None]:
        """流式生成文本"""
        try:
            stream = await self.client.chat.completions.create(
                model=model,
                messages=[{"role": "user", "content": prompt}],
                max_tokens=max_tokens,
                temperature=temperature,
                stream=True,
                **kwargs
            )

            async for chunk in stream:
                if chunk.choices[0].delta.content:
                    yield chunk.choices[0].delta.content
        except Exception as e:
            logger.error(f"OpenAI 流式生成失败: {e}")
            raise
```

### 3. SSE 流式响应

```python
from fastapi.responses import StreamingResponse
from typing import AsyncGenerator
import json

async def generate_sse_stream(
    provider: BaseAIProvider,
    prompt: str,
    max_tokens: int = 1000
) -> AsyncGenerator[str, None]:
    """生成 SSE 格式的流式数据"""
    try:
        # 发送开始事件
        yield f"data: {json.dumps({'type': 'start'})}\n\n"

        # 流式生成内容
        async for chunk in provider.generate_stream(prompt, max_tokens):
            data = {
                "type": "content",
                "content": chunk
            }
            yield f"data: {json.dumps(data)}\n\n"

        # 发送完成事件
        yield f"data: {json.dumps({'type': 'done'})}\n\n"
    except Exception as e:
        # 发送错误事件
        error_data = {
            "type": "error",
            "message": str(e)
        }
        yield f"data: {json.dumps(error_data)}\n\n"

@router.post("/generate-stream")
async def generate_stream_endpoint(
    request: GenerateRequest,
    user_id: str = Depends(get_current_user)
):
    """SSE 流式生成接口"""
    provider = get_ai_provider()  # 获取 AI 提供商

    return StreamingResponse(
        generate_sse_stream(provider, request.prompt, request.max_tokens),
        media_type="text/event-stream"
    )
```

```python
class AppException(Exception):
    """应用基础异常"""
    def __init__(self, message: str, status_code: int = 500):
        self.message = message
        self.status_code = status_code
        super().__init__(self.message)

class NotFoundException(AppException):
    """资源不存在异常"""

class ValidationException(AppException):
```

### 1. 日志配置

```python
import logging
from logging.handlers import RotatingFileHandler
import sys

def setup_logger(name: str) -> logging.Logger:
    """配置日志记录器"""
    logger = logging.getLogger(name)
    logger.setLevel(logging.INFO)

    # 控制台处理器
    console_handler = logging.StreamHandler(sys.stdout)
    console_handler.setLevel(logging.INFO)
    console_formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    console_handler.setFormatter(console_formatter)

    # 文件处理器
    file_handler = RotatingFileHandler(
        'logs/app.log',
        maxBytes=10*1024*1024,  # 10MB
        backupCount=5
    )
    file_handler.setLevel(logging.INFO)
    file_formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    file_handler.setFormatter(file_formatter)

    logger.addHandler(console_handler)
    logger.addHandler(file_handler)

    return logger
```

### 2. 缓存

```python
from functools import lru_cache
from typing import Optional
import redis.asyncio as redis

# Redis 缓存
class CacheService:
    def __init__(self):
        self.redis = redis.from_url("redis://localhost:6379")

    async def get(self, key: str) -> Optional[str]:
        """获取缓存"""
        return await self.redis.get(key)

    async def set(self, key: str, value: str, expire: int = 3600):
        """设置缓存"""
        await self.redis.set(key, value, ex=expire)

    async def delete(self, key: str):
        """删除缓存"""
        await self.redis.delete(key)

# 使用缓存
cache = CacheService()

async def get_project_cached(db: AsyncSession, project_id: str):
    """获取项目（带缓存）"""
    cache_key = f"project:{project_id}"

    # 尝试从缓存获取
    cached = await cache.get(cache_key)
    if cached:
        return json.loads(cached)

    # 从数据库查询
    project = await db.get(Project, project_id)
    if project:
        # 存入缓存
        await cache.set(cache_key, json.dumps(project.to_dict()))

    return project
```
